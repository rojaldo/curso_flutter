=== Introducción a la Persistencia

==== ¿Qué es la Persistencia de Datos?

La persistencia de datos es la capacidad de almacenar información de
forma que sobreviva a cierres de la aplicación, reinicios del
dispositivo o actualizaciones de la app.

....
┌─────────────────────────────────────────────────────────────┐
│              Tipos de Persistencia en Flutter              │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│  SharedPreferences                                          │
│  ├── Datos simples (clave-valor)                           │
│  ├── Ideal para preferencias, configuraciones              │
│  ├── Almacenamiento: JSON en disco                         │
│  └── Límite: ~100KB recomendado                             │
│                                                             │
│  Hive                                                       │
│  ├── Base de datos NoSQL ligera                            │
│  ├── Ideal para objetos, cache                              │
│  ├── Rápido, sin dependencias nativas                      │
│  └── Cifrado opcional                                       │
│                                                             │
│  SQLite (sqflite)                                           │
│  ├── Base de datos relacional                              │
│  ├── Ideal para datos estructurados, relaciones            │
│  ├── Consultas SQL completas                                │
│  └── Amplio soporte, bien documentado                       │
│                                                             │
│  Drift (ORM)                                                │
│  ├── ORM sobre SQLite                                       │
│  ├── Tipado seguro, migraciones automáticas                 │
│  ├── Consultas tipadas, reactive                            │
│  └── Ideal para apps complejas                              │
│                                                             │
│  Isar                                                       │
│  ├── Base de datos NoSQL muy rápida                        │
│  ├── Ideal para grandes volúmenes                           │

│                                                             │
│  Secure Storage                                              │

==== Criterios de Selección

[source,dart]
----
class StorageSelectionGuide {
  static String recommend({
    required bool needsEncryption,
    required bool needsRelations,
    required int dataSizeKB,
    required bool needsSync,
    required bool needsQuerying,
  }) {
    // Necesita cifrado alto
    if (needsEncryption && dataSizeKB < 100) {
      return 'Flutter Secure Storage';
    }
    
    // Datos simples
    if (dataSizeKB < 100 && !needsRelations && !needsQuerying) {
      return 'SharedPreferences';
    }
    
    // Necesita relaciones
    if (needsRelations) {
      return dataSizeKB > 10000 ? 'Drift' : 'sqflite';
    }
    
    // Necesita consultas complejas
    if (needsQuerying && needsSync) {
      return 'Isar';
    }
    
    // Datos medianos sin relaciones
    if (dataSizeKB < 10000 && !needsRelations) {
      return 'Hive';
    }
    
    // Grandes volúmenes
    return 'Isar';
  }
}

    return await _prefs.setInt(key, value);
  }

    return _prefs.getInt(key);

    return await _prefs.setDouble(key, value);
  }

    return _prefs.getDouble(key);
  }
  
  // Bool
  Future<bool> setBool(String key, bool value) async {
    return await _prefs.setBool(key, value);
  }
  
  bool? getBool(String key) {
    return _prefs.getBool(key);
  }
  
  // List<String>
  Future<bool> setStringList(String key, List<String> value) async {
    return await _prefs.setStringList(key, value);

    return _prefs.getStringList(key);

    return _prefs.containsKey(key);

    return await _prefs.clear();

class PreferencesManager {
  static const String _themeKey = 'app_theme';
  static const String _localeKey = 'app_locale';
  static const String _notificationsKey = 'notifications_enabled';
  static const String _onboardingKey = 'onboarding_completed';
  static const String _lastSyncKey = 'last_sync_timestamp';
  static const String _recentSearchesKey = 'recent_searches';
  static const String _userTokenKey = 'user_token';
  
  final SharedPreferencesService _service;
  
  PreferencesManager(this._service);
  
  // ========== Tema ==========
  
  Future<void> setTheme(ThemeMode mode) async {
    await _service.setString(_themeKey, mode.name);
  }
  
  ThemeMode getTheme() {
    final name = _service.getString(_themeKey);
    return ThemeMode.values.firstWhere(
      (mode) => mode.name == name,
      orElse: () => ThemeMode.system,
    );
  }
  
  // ========== Idioma ==========
  
  Future<void> setLocale(Locale locale) async {
    await _service.setString(_localeKey, locale.languageCode);
  }
  
  Locale getLocale() {
    final code = _service.getString(_localeKey);
    return code != null ? Locale(code) : const Locale('es');
  }
  
  // ========== Notificaciones ==========
  
  Future<void> setNotificationsEnabled(bool enabled) async {
    await _service.setBool(_notificationsKey, enabled);
  }
  
  bool getNotificationsEnabled() {
    return _service.getBool(_notificationsKey) ?? true;
  }
  
  // ========== Onboarding ==========
  
  Future<void> setOnboardingCompleted(bool completed) async {
    await _service.setBool(_onboardingKey, completed);
  }
  
  bool getOnboardingCompleted() {
    return _service.getBool(_onboardingKey) ?? false;
  }
  
  // ========== Última sincronización ==========
  
  Future<void> setLastSync(DateTime dateTime) async {
    await _service.setInt(_lastSyncKey, dateTime.millisecondsSinceEpoch);
  }
  
  DateTime? getLastSync() {
    final timestamp = _service.getInt(_lastSyncKey);
    return timestamp != null ? DateTime.fromMillisecondsSinceEpoch(timestamp) : null;
  }
  
  // ========== Búsquedas recientes ==========
  
  Future<void> addRecentSearch(String search) async {
    final searches = getRecentSearches();
    searches.remove(search); // Evitar duplicados
    searches.insert(0, search);
    
    // Mantener solo las últimas 10
    if (searches.length > 10) {
      searches.removeRange(10, searches.length);
    }
    
    await _service.setStringList(_recentSearchesKey, searches);
  }
  
  List<String> getRecentSearches() {
    return _service.getStringList(_recentSearchesKey) ?? [];
  }
  
  Future<void> clearRecentSearches() async {
    await _service.remove(_recentSearchesKey);
  }
  
  // ========== Token (no seguro, solo para demo) ==========
  
  Future<void> setUserToken(String token) async {
    await _service.setString(_userTokenKey, token);
  }
  
  String? getUserToken() {
    return _service.getString(_userTokenKey);
  }
  
  Future<void> removeUserToken() async {
    await _service.remove(_userTokenKey);
  }
}

class HiveUserService {
  late Box<User> _userBox;
  
  Future<void> init() async {
    _userBox = await Hive.openBox<User>('users');
  }
  
  // ========== CREATE ==========
  
  Future<void> addUser(User user) async {
    await _userBox.put(user.id, user);
  }
  
  Future<void> addUsers(List<User> users) async {
    final Map<String, User> userMap = {
      for (var user in users) user.id: user
    };
    await _userBox.putAll(userMap);
  }
  
  // ========== READ ==========
  
  User? getUser(String id) {
    return _userBox.get(id);
  }
  
  List<User> getAllUsers() {
    return _userBox.values.toList();
  }
  
  List<User> getActiveUsers() {
    return _userBox.values.where((user) => user.isActive).toList();
  }
  
  List<User> searchUsers(String query) {
    final lowerQuery = query.toLowerCase();
    return _userBox.values.where((user) {
      return user.name.toLowerCase().contains(lowerQuery) ||
             user.email.toLowerCase().contains(lowerQuery);
    }).toList();
  }
  
  // ========== UPDATE ==========
  
  Future<void> updateUser(User user) async {
    await user.save(); // HiveObject.save()
  }
  
  Future<void> updateUserFields(String id, {
    String? name,
    String? email,
    int? age,
    bool? isActive,
  }) async {
    final user = _userBox.get(id);
    if (user != null) {
      if (name != null) user.name = name;
      if (email != null) user.email = email;
      if (age != null) user.age = age;
      if (isActive != null) user.isActive = isActive;
      await user.save();
    }
  }
  
  // ========== DELETE ==========
  
  Future<void> deleteUser(String id) async {
    await _userBox.delete(id);
  }
  
  Future<void> deleteAllUsers() async {
    await _userBox.clear();
  }
  
  // ========== UTILIDADES ==========
  
  int getUserCount() {
    return _userBox.length;
  }
  
  bool containsUser(String id) {
    return _userBox.containsKey(id);
  }
  
  Iterable<String> getAllKeys() {
    return _userBox.keys;
  }
  
  // ========== BATCH OPERATIONS ==========
  
  Future<void> batchUpdate(List<User> users) async {
    await _userBox.putAll({
      for (var user in users) user.id: user
    });
  }
  
  Future<void> batchDelete(List<String> ids) async {
    for (var id in ids) {
      await _userBox.delete(id);
    }
  }
}

class HiveQueryService {
  final Box<User> _userBox;
  
  HiveQueryService(this._userBox);
  
  // Filtrar por edad
  List<User> getUsersByAgeRange(int minAge, int maxAge) {
    return _userBox.values.where((user) {
      return user.age >= minAge && user.age <= maxAge;
    }).toList();
  }
  
  // Filtrar por tags
  List<User> getUsersByTag(String tag) {
    return _userBox.values.where((user) {
      return user.tags.contains(tag);
    }).toList();
  }
  
  // Ordenar por nombre
  List<User> getUsersSortedByName({bool ascending = true}) {
    final users = _userBox.values.toList();
    users.sort((a, b) => ascending
        ? a.name.compareTo(b.name)
        : b.name.compareTo(a.name));
    return users;
  }
  
  // Ordenar por fecha de creación
  List<User> getUsersSortedByDate({bool ascending = true}) {
    final users = _userBox.values.toList();
    users.sort((a, b) => ascending
        ? a.createdAt.compareTo(b.createdAt)
        : b.createdAt.compareTo(a.createdAt));
    return users;
  }
  
  // Paginación
  List<User> getUsersPaginated({int page = 0, int pageSize = 20}) {
    final users = _userBox.values.toList();
    final start = page * pageSize;
    final end = start + pageSize;
    
    if (start >= users.length) return [];
    
    return users.sublist(
      start,
      end.clamp(0, users.length),
    );
  }
  
  // Búsqueda avanzada
  List<User> advancedSearch({
    String? name,
    String? email,
    int? minAge,
    int? maxAge,
    bool? isActive,
    String? tag,
  }) {
    return _userBox.values.where((user) {
      if (name != null && !user.name.toLowerCase().contains(name.toLowerCase())) {
        return false;
      }
      if (email != null && !user.email.toLowerCase().contains(email.toLowerCase())) {
        return false;
      }
      if (minAge != null && user.age < minAge) {
        return false;
      }
      if (maxAge != null && user.age > maxAge) {
        return false;
      }
      if (isActive != null && user.isActive != isActive) {
        return false;
      }
      if (tag != null && !user.tags.contains(tag)) {
        return false;
      }
      return true;
    }).toList();
  }
  
  // Contar
  int countActiveUsers() {
    return _userBox.values.where((user) => user.isActive).length;
  }
  
  // Agrupar por tag
  Map<String, List<User>> groupUsersByTag() {
    final Map<String, List<User>> groups = {};
    
    for (var user in _userBox.values) {
      for (var tag in user.tags) {
        groups.putIfAbsent(tag, () => []);
        groups[tag]!.add(user);
      }
    }
    
    return groups;
  }
}

class SecureHiveService {
  static late Box _secureBox;
  static late Key _encryptionKey;
  
  // Inicializar con clave de cifrado

    await _secureBox.put(key, data);

    return _secureBox.get(key);

    await _secureBox.delete(key);
  }

}

      }
    }

  int getUserCount() {
    return _lazyBox.length;
  }
  
  // Verificar existencia

  }

import 'package:sqflite/sqflite.dart';
import 'package:path/path.dart';

class DatabaseHelper {
  static final DatabaseHelper _instance = DatabaseHelper._internal();
  factory DatabaseHelper() => _instance;
  DatabaseHelper._internal();
  
  static Database? _database;
  
  Future<Database> get database async {
    if (_database != null) return _database!;
    _database = await _initDatabase();
    return _database!;
  }
  
  Future<Database> _initDatabase() async {
    final dbPath = await getDatabasesPath();
    final path = join(dbPath, 'app_database.db');
    
    return await openDatabase(
      path,
      version: 1,
      onCreate: _onCreate,
      onUpgrade: _onUpgrade,
      onDowngrade: _onDowngrade,
    );
  }
  
  Future<void> _onCreate(Database db, int version) async {
    // Crear tablas
    await db.execute('''
      CREATE TABLE users (
        id TEXT PRIMARY KEY,
        name TEXT NOT NULL,
        email TEXT NOT NULL UNIQUE,
        age INTEGER,
        is_active INTEGER DEFAULT 1,
        created_at TEXT NOT NULL,
        updated_at TEXT NOT NULL
      )
    ''');
    
    await db.execute('''
      CREATE TABLE tasks (
        id TEXT PRIMARY KEY,
        title TEXT NOT NULL,
        description TEXT,
        status TEXT NOT NULL DEFAULT 'pending',
        priority INTEGER DEFAULT 0,
        user_id TEXT NOT NULL,
        created_at TEXT NOT NULL,
        due_date TEXT,
        FOREIGN KEY (user_id) REFERENCES users (id)
          ON DELETE CASCADE
      )
    ''');
    
    await db.execute('''
      CREATE TABLE tags (
        id TEXT PRIMARY KEY,
        name TEXT NOT NULL UNIQUE,
        color TEXT
      )
    ''');
    
    await db.execute('''
      CREATE TABLE task_tags (
        task_id TEXT NOT NULL,
        tag_id TEXT NOT NULL,
        PRIMARY KEY (task_id, tag_id),
        FOREIGN KEY (task_id) REFERENCES tasks (id)
          ON DELETE CASCADE,
        FOREIGN KEY (tag_id) REFERENCES tags (id)
          ON DELETE CASCADE
      )
    ''');
    
    // Crear índices
    await db.execute('CREATE INDEX idx_tasks_user ON tasks (user_id)');
    await db.execute('CREATE INDEX idx_tasks_status ON tasks (status)');
    await db.execute('CREATE INDEX idx_tasks_due_date ON tasks (due_date)');
  }
  
  Future<void> _onUpgrade(Database db, int oldVersion, int newVersion) async {
    // Migraciones
    if (oldVersion < 2) {
      // Añadir columna
      await db.execute('ALTER TABLE users ADD COLUMN avatar TEXT');
    }
    
    if (oldVersion < 3) {
      // Crear nueva tabla
      await db.execute('''

        SUM(CASE WHEN t.status = 'pending' THEN 1 ELSE 0 END) as pending_tasks,
        AVG(t.priority) as avg_priority

      LEFT JOIN tasks t ON u.id = t.user_id

      whereArgs = [status];

    final offset = page * pageSize;
    
    final results = await db.query(
      'tasks',
      where: whereClause.isEmpty ? null : whereClause,
      whereArgs: whereArgs.isEmpty ? null : whereArgs,
      orderBy: 'created_at DESC',
      limit: pageSize,
      offset: offset,
    );
    
    return results.map((map) => Task.fromMap(map)).toList();
  }

class TokenManager {
  static const String _accessTokenKey = 'access_token';
  static const String _refreshTokenKey = 'refresh_token';
  static const String _tokenExpiryKey = 'token_expiry';
  static const String _userIdKey = 'user_id';
  
  final SecureStorageService _storage;
  
  TokenManager(this._storage);
  
  // Guardar tokens
  Future<void> saveTokens({
    required String accessToken,
    required String refreshToken,
    required int expiresInSeconds,
    String? userId,
  }) async {
    final expiry = DateTime.now().add(Duration(seconds: expiresInSeconds));
    
    await Future.wait([
      _storage.write(_accessTokenKey, accessToken),
      _storage.write(_refreshTokenKey, refreshToken),
      _storage.write(_tokenExpiryKey, expiry.toIso8601String()),
      if (userId != null) _storage.write(_userIdKey, userId),
    ]);
  }
  
  // Obtener access token
  Future<String?> getAccessToken() async {
    final expiry = await _storage.read(_tokenExpiryKey);
    if (expiry == null) return null;
    
    final expiryDate = DateTime.parse(expiry);
    if (DateTime.now().isAfter(expiryDate)) {
      // Token expirado
      return null;
    }
    
    return await _storage.read(_accessTokenKey);
  }
  
  // Obtener refresh token
  Future<String?> getRefreshToken() async {
    return await _storage.read(_refreshTokenKey);
  }
  
  // Verificar si hay sesión
  Future<bool> hasSession() async {
    final accessToken = await getAccessToken();
    final refreshToken = await getRefreshToken();
    return accessToken != null || refreshToken != null;
  }
  
  // Verificar si token está expirado
  Future<bool> isTokenExpired() async {
    final expiry = await _storage.read(_tokenExpiryKey);
    if (expiry == null) return true;
    
    return DateTime.now().isAfter(DateTime.parse(expiry));
  }
  
  // Limpiar tokens
  Future<void> clearTokens() async {
    await Future.wait([
      _storage.delete(_accessTokenKey),
      _storage.delete(_refreshTokenKey),
      _storage.delete(_tokenExpiryKey),
      _storage.delete(_userIdKey),
    ]);
  }
  
  // Obtener user ID
  Future<String?> getUserId() async {
    return await _storage.read(_userIdKey);
  }
}

class StorageRecommendation {
  static String recommend({
    required bool needsEncryption,
    required bool needsRelations,
    required bool needsComplexQueries,
    required int estimatedSizeKB,
    required bool needsRealTimeUpdates,
    required bool isSensitive,
  }) {
    // Datos sensibles pequeños
    if (isSensitive && estimatedSizeKB < 10) {
      return 'Flutter Secure Storage';
    }
    
    // Datos simples y pequeños
    if (estimatedSizeKB < 100 && !needsRelations && !needsComplexQueries) {
      return 'SharedPreferences';
    }
    
    // Necesita relaciones y consultas complejas
    if (needsRelations && needsComplexQueries) {
      return estimatedSizeKB > 1000 ? 'Drift' : 'SQLite';
    }
    
    // Necesita actualizaciones en tiempo real
    if (needsRealTimeUpdates) {
      return needsRelations ? 'Drift' : 'Isar';
    }
    
    // Datos medianos sin relaciones
    if (estimatedSizeKB < 10000) {
      return needsEncryption ? 'Hive (encrypted)' : 'Hive';
    }
    
    // Grandes volúmenes
    return 'Isar';
  }
}

class CacheStrategy {
  final SharedPreferencesService _prefs;
  final HiveUserService _hive;
  
  CacheStrategy(this._prefs, this._hive);
  
  // Caché de dos niveles: memoria -> disco
  final Map<String, dynamic> _memoryCache = {};
  
  Future<T?> get<T>(String key, {
    required Future<T?> Function() fetchFromNetwork,
    Duration cacheDuration = const Duration(hours: 1),
    bool forceRefresh = false,
  }) async {
    // Nivel 1: Memoria
    if (!forceRefresh && _memoryCache.containsKey(key)) {
      final entry = _memoryCache[key] as _CacheEntry;
      if (!entry.isExpired) {
        return entry.data as T;
      }
    }
    
    // Nivel 2: Disco (Hive)
    if (!forceRefresh) {
      final cached = await _hive.getCached<T>(key);
      if (cached != null) {
        _memoryCache[key] = _CacheEntry(data: cached, expiry: DateTime.now().add(cacheDuration));
        return cached;
      }
    }
    
    // Nivel 3: Red
    final data = await fetchFromNetwork();
    if (data != null) {
      // Guardar en caché
      await _hive.cache(key, data, cacheDuration);
      _memoryCache[key] = _CacheEntry(data: data, expiry: DateTime.now().add(cacheDuration));
    }
    
    return data;
  }
  
  void invalidate(String key) {
    _memoryCache.remove(key);
    _hive.invalidateCache(key);
  }
  
  void invalidateAll() {
    _memoryCache.clear();
    _hive.invalidateAllCache();
  }
}

class _CacheEntry {
  final dynamic data;
  final DateTime expiry;
  
  _CacheEntry({required this.data, required this.expiry});
  
  bool get isExpired => DateTime.now().isAfter(expiry);
}

  final TaskRepository _repository;
  
  TaskListNotifier(this._repository) : super([]) {
    _loadTasks();
  }
  
  Future<void> _loadTasks() async {
    state = await _repository.getAllTasks();
  }
  
  Future<void> addTask(TaskHive task) async {
    await _repository.addTask(task);
    state = [...state, task];
  }
  
  Future<void> updateTask(TaskHive task) async {
    await _repository.updateTask(task);
    state = state.map((t) => t.id == task.id ? task : t).toList();
  }
  
  Future<void> deleteTask(String id) async {
    await _repository.deleteTask(id);
    state = state.where((t) => t.id != id).toList();
  }
  
  Future<void> toggleTask(String id) async {
    final task = state.firstWhere((t) => t.id == id);
    final updated = TaskHive(
      id: task.id,
      title: task.title,
      description: task.description,
      isCompleted: !task.isCompleted,
      categoryId: task.categoryId,
      createdAt: task.createdAt,
      dueDate: task.dueDate,
      priority: task.priority,
    );
    await updateTask(updated);
  }
}

=== Conclusión

La persistencia de datos en Flutter ofrece múltiples soluciones según
las necesidades:

[arabic]
. *SharedPreferences*: Ideal para datos simples (preferencias,
configuraciones)
. *Hive*: NoSQL ligero para objetos, rápido y fácil de usar
. *SQLite*: Relacional completo para datos estructurados con relaciones
. *Drift*: ORM sobre SQLite con tipado seguro y consultas reactivas
. *Isar*: NoSQL muy rápido para grandes volúmenes con consultas
avanzadas
. *Secure Storage*: Cifrado para datos sensibles (tokens, credenciales)

*Recomendaciones:* - Usar repositorios para abstraer la capa de datos -
Implementar caché en múltiples niveles para rendimiento - Cifrar datos
sensibles con Flutter Secure Storage - Usar Drift para apps con
relaciones complejas - Usar Isar para grandes volúmenes de datos -
Implementar migraciones para bases de datos relacionales
