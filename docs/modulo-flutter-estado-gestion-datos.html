Introducción al Estado

¿Qué es el Estado?

El estado en Flutter es la información que un widget necesita para renderizarse. Cuando el estado cambia, el widget se reconstruye para reflejar esos cambios.

┌─────────────────────────────────────────┐
│              Tipos de Estado            │
├─────────────────────────────────────────┤
│                                         │
│  Estado Local (Ephemeral)               │
│  ├── StatefulWidget + setState          │
│  ├── Solo afecta al widget actual       │
│  └── Ejemplo: contador, selección       │
│                                         │
│  Estado de Aplicación (App State)      │
│  ├── Compartido entre widgets           │
│  ├── Requiere gestión externa          │
│  └── Ejemplo: auth, carrito, usuario    │
│                                         │
│  Estado de Widget                       │
│  ├── Inmutable en StatelessWidget       │
│  ├── Mutable en StatefulWidget          │
│  └── Se pasa vía constructor            │
│                                         │
└─────────────────────────────────────────┘

setState - Fundamentos

Concepto Básico

setState es el método más simple para gestionar estado local en Flutter. Marca el widget como “sucio” y programa una reconstrucción.

class CounterExample extends StatefulWidget {
  const CounterExample({super.key});

  @override
  State<CounterExample> createState() => _CounterExampleState();
}

class _CounterExampleState extends State<CounterExample> {
  int _counter = 0;

  void _increment() {
    setState(() {
      _counter++;
    });
  }

  void _decrement() {
    setState(() {
      _counter--;
    });
  }

  void _reset() {
    setState(() {
      _counter = 0;
    });
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: const Text('Contador con setState')),
      body: Center(
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            const Text('Valor actual:'),
            Text(
              '$_counter',
              style: Theme.of(context).textTheme.headlineLarge,
            ),
          ],
        ),
      ),
      floatingActionButton: Column(
        mainAxisAlignment: MainAxisAlignment.end,
        children: [
          FloatingActionButton(
            heroTag: 'increment',
            onPressed: _increment,
            child: const Icon(Icons.add),
          ),
          const SizedBox(height: 8),
          FloatingActionButton(
            heroTag: 'decrement',
            onPressed: _decrement,
            child: const Icon(Icons.remove),
          ),
          const SizedBox(height: 8),
          FloatingActionButton(
            heroTag: 'reset',
            onPressed: _reset,
            child: const Icon(Icons.refresh),
          ),
        ],
      ),
    );
  }
}

Reglas de setState

class SetStateRules extends StatefulWidget {
  const SetStateRules({super.key});

  @override
  State<SetStateRules> createState() => _SetStateRulesState();
}

class _SetStateRulesState extends State<SetStateRules> {
  int _counter = 0;
  String _status = 'Inicial';
  List<String> _items = [];

  // ✅ Correcto: Actualizar estado dentro de setState
  void _incrementCorrect() {
    setState(() {
      _counter++;
      _status = 'Incrementado a $_counter';
    });
  }

  // ❌ Incorrecto: setState con valor hardcodeado
  void _wrongUsage() {
    // Esto actualiza pero puede causar confusión
    setState(() {
      _counter = _counter + 1; // No usar el valor anterior directamente
    });
  }

  // ✅ Correcto: setState asíncrono con callbacks
  Future<void> _fetchData() async {
    setState(() {
      _status = 'Cargando...';
    });

    try {
      final data = await _fetchItemsFromApi();
      if (mounted) { // Verificar que el widget sigue montado
        setState(() {
          _items = data;
          _status = 'Cargado';
        });
      }
    } catch (e) {
      if (mounted) {
        setState(() {
          _status = 'Error: $e';
        });
      }
    }
  }

  // ✅ Correcto: Lógica de cálculo fuera de setState
  void _calculateTotal() {
    final total = _items.length * 10; // Cálculo fuera
    setState(() {
      _status = 'Total: $total';
    });
  }

  // ❌ Incorrecto: setState en initState
  @override
  void initState() {
    super.initState();
    // setState(() {}); // ❌ Error: No llamar setState en initState
    // Usar:
    WidgetsBinding.instance.addPostFrameCallback((_) {
      // Aquí sí se puede usar setState
    });
  }

  // ❌ Incorrecto: setState en build
  @override
  Widget build(BuildContext context) {
    // setState(() {}); // ❌ Error: Causa loop infinito
    return Container();
  }

  // ✅ Correcto: Verificar mounted antes de setState
  void _updateFromCallback() {
    if (mounted) {
      setState(() {
        _status = 'Actualizado';
      });
    }
  }

  Future<List<String>> _fetchItemsFromApi() async {
    await Future.delayed(const Duration(seconds: 1));
    return ['Item 1', 'Item 2', 'Item 3'];
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: const Text('Reglas de setState')),
      body: Center(
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            Text('Contador: $_counter'),
            Text('Estado: $_status'),
            Text('Items: ${_items.length}'),
          ],
        ),
      ),
    );
  }
}

class FormExample extends StatefulWidget {
  const FormExample({super.key});

  @override
  State<FormExample> createState() => _FormExampleState();
}

class _FormExampleState extends State<FormExample> {
  final _formKey = GlobalKey<FormState>();

  // Estado del formulario
  String _name = '';
  String _email = '';
  String _phone = '';
  String _selectedCountry = 'España';
  bool _acceptTerms = false;
  bool _subscribeNewsletter = false;
  List<String> _selectedInterests = [];

  // Estado de UI
  bool _isSubmitting = false;
  String? _errorMessage;
  bool _showSuccess = false;

  final List<String> _countries = ['España', 'México', 'Argentina', 'Colombia', 'Chile'];
  final List<String> _interests = ['Tecnología', 'Deportes', 'Música', 'Viajes', 'Gastronomía'];

  void _toggleInterest(String interest) {
    setState(() {
      if (_selectedInterests.contains(interest)) {
        _selectedInterests.remove(interest);
      } else {
        _selectedInterests.add(interest);
      }
    });
  }

  Future<void> _submitForm() async {
    if (!_formKey.currentState!.validate()) return;

    setState(() {
      _isSubmitting = true;
      _errorMessage = null;
    });

    try {
      // Simular envío
      await Future.delayed(const Duration(seconds: 2));

      // Procesar datos
      _formKey.currentState!.save();

      setState(() {
        _isSubmitting = false;
        _showSuccess = true;
      });

      // Resetear después de mostrar éxito
      await Future.delayed(const Duration(seconds: 2));

      if (mounted) {
        setState(() {
          _showSuccess = false;
          _name = '';
          _email = '';
          _phone = '';
          _acceptTerms = false;
          _subscribeNewsletter = false;
          _selectedInterests = [];
        });
        _formKey.currentState!.reset();
      }
    } catch (e) {
      if (mounted) {
        setState(() {
          _isSubmitting = false;
          _errorMessage = 'Error al enviar: $e';
        });
      }
    }
  }

  @override
  Widget build(BuildContext context) {
    if (_showSuccess) {
      return Scaffold(
        appBar: AppBar(title: const Text('Formulario')),
        body: const Center(
          child: Column(
            mainAxisAlignment: MainAxisAlignment.center,
            children: [
              Icon(Icons.check_circle, color: Colors.green, size: 64),
              SizedBox(height: 16),
              Text('¡Enviado correctamente!', style: TextStyle(fontSize: 20)),
            ],
          ),
        ),
      );
    }

    return Scaffold(
      appBar: AppBar(title: const Text('Formulario Complejo')),
      body: Form(
        key: _formKey,
        child: SingleChildScrollView(
          padding: const EdgeInsets.all(16),
          child: Column(
            crossAxisAlignment: CrossAxisAlignment.stretch,
            children: [
              // Campo nombre
              TextFormField(
                decoration: const InputDecoration(
                  labelText: 'Nombre',
                  border: OutlineInputBorder(),
                ),
                validator: (value) {
                  if (value == null || value.isEmpty) {
                    return 'Por favor introduce tu nombre';
                  }
                  if (value.length < 3) {
                    return 'El nombre debe tener al menos 3 caracteres';
                  }
                  return null;
                },
                onSaved: (value) => _name = value ?? '',
              ),
              const SizedBox(height: 16),

              // Campo email
              TextFormField(
                decoration: const InputDecoration(
                  labelText: 'Email',
                  border: OutlineInputBorder(),
                ),
                keyboardType: TextInputType.emailAddress,
                validator: (value) {
                  if (value == null || value.isEmpty) {
                    return 'Por favor introduce tu email';
                  }
                  if (!value.contains('@')) {
                    return 'Por favor introduce un email válido';
                  }
                  return null;
                },
                onSaved: (value) => _email = value ?? '',
              ),
              const SizedBox(height: 16),

              // Campo teléfono
              TextFormField(
                decoration: const InputDecoration(
                  labelText: 'Teléfono',
                  border: OutlineInputBorder(),
                ),
                keyboardType: TextInputType.phone,
                onSaved: (value) => _phone = value ?? '',
              ),
              const SizedBox(height: 16),

              // Dropdown país
              DropdownButtonFormField<String>(
                value: _selectedCountry,
                decoration: const InputDecoration(
                  labelText: 'País',
                  border: OutlineInputBorder(),
                ),
                items: _countries.map((country) {
                  return DropdownMenuItem(
                    value: country,
                    child: Text(country),
                  );
                }).toList(),
                onChanged: (value) {
                  setState(() {
                    _selectedCountry = value ?? 'España';
                  });
                },
              ),
              const SizedBox(height: 16),

              // Intereses
              const Text('Intereses:', style: TextStyle(fontWeight: FontWeight.bold)),
              Wrap(
                spacing: 8,
                children: _interests.map((interest) {
                  final isSelected = _selectedInterests.contains(interest);
                  return FilterChip(
                    label: Text(interest),
                    selected: isSelected,
                    onSelected: (_) => _toggleInterest(interest),
                  );
                }).toList(),
              ),
              const SizedBox(height: 16),

              // Checkboxes
              CheckboxListTile(
                title: const Text('Acepto los términos y condiciones'),
                value: _acceptTerms,
                onChanged: (value) {
                  setState(() {
                    _acceptTerms = value ?? false;
                  });
                },
              ),
              CheckboxListTile(
                title: const Text('Suscribirse al newsletter'),
                value: _subscribeNewsletter,
                onChanged: (value) {
                  setState(() {
                    _subscribeNewsletter = value ?? false;
                  });
                },
              ),
              const SizedBox(height: 16),

              // Error message
              if (_errorMessage != null)
                Container(
                  padding: const EdgeInsets.all(8),
                  decoration: BoxDecoration(
                    color: Colors.red.shade100,
                    borderRadius: BorderRadius.circular(8),
                  ),
                  child: Text(
                    _errorMessage!,
                    style: const TextStyle(color: Colors.red),
                  ),
                ),
              const SizedBox(height: 16),

              // Botón de envío
              ElevatedButton(
                onPressed: _isSubmitting || !_acceptTerms ? null : _submitForm,
                child: _isSubmitting
                    ? const SizedBox(
                        width: 20,
                        height: 20,
                        child: CircularProgressIndicator(strokeWidth: 2),
                      )
                    : const Text('Enviar'),
              ),
            ],
          ),
        ),
      ),
    );
  }
}

InheritedWidget - Estado Heredado

Concepto

InheritedWidget permite propagar datos hacia abajo en el árbol de widgets sin pasarlos explícitamente por cada constructor.

// Widget heredado personalizado
class CounterInherited extends InheritedWidget {
  final int counter;
  final VoidCallback increment;
  final VoidCallback decrement;

  const CounterInherited({
    super.key,
    required this.counter,
    required this.increment,
    required this.decrement,
    required super.child,
  });

  // Método estático para acceder desde descendientes
  static CounterInherited of(BuildContext context) {
    final CounterInherited? result =
        context.dependOnInheritedWidgetOfExactType<CounterInherited>();
    assert(result != null, 'No CounterInherited found in context');
    return result!;
  }

  @override
  bool updateShouldNotify(CounterInherited oldWidget) {
    return counter != oldWidget.counter;
  }
}

class UserModel {

  final String name;
  final String email;
  final String? avatar;

  const UserModel({
    required this.id,

    required this.email,
    this.avatar,

    String? name,

    String? avatar,

      id: id ?? this.id,
      name: name ?? this.name,
      email: email ?? this.email,
      avatar: avatar ?? this.avatar,

  }

// InheritedWidget para autenticación
class AuthInherited extends InheritedWidget {
  final AuthState state;
  final Future<void> Function(String email, String password) login;
  final Future<void> Function(String name, String email, String password) register;
  final VoidCallback logout;

  const AuthInherited({
    super.key,
    required this.state,
    required this.login,
    required this.register,
    required this.logout,
    required super.child,
  });

  static AuthInherited of(BuildContext context) {
    final result = context.dependOnInheritedWidgetOfExactType<AuthInherited>();
    assert(result != null, 'No AuthInherited found in context');
    return result!;
  }

  static AuthInherited? maybeOf(BuildContext context) {
    return context.dependOnInheritedWidgetOfExactType<AuthInherited>();
  }

  @override
  bool updateShouldNotify(AuthInherited oldWidget) {
    return state.isAuthenticated != oldWidget.state.isAuthenticated ||
           state.user != oldWidget.state.user ||
           state.isLoading != oldWidget.state.isLoading ||
           state.error != oldWidget.state.error;
  }
}

// Provider de autenticación
class AuthProvider extends StatefulWidget {
  final Widget child;

  const AuthProvider({super.key, required this.child});

  @override
  State<AuthProvider> createState() => AuthProviderState();
}

class AuthProviderState extends State<AuthProvider> {
  AuthState _state = const AuthState();

  AuthState get state => _state;

  Future<void> login(String email, String password) async {
    setState(() {
      _state = _state.copyWith(isLoading: true, error: null);
    });

    try {
      // Simular llamada API
      await Future.delayed(const Duration(seconds: 1));

      setState(() {
        _state = AuthState(
          user: UserModel(
            id: '1',
            name: 'Usuario Demo',
            email: email,
          ),
          isLoading: false,
          isAuthenticated: true,
        );
      });
    } catch (e) {
      setState(() {
        _state = _state.copyWith(
          isLoading: false,
          error: 'Error de autenticación: $e',
        );
      });
    }
  }

  Future<void> register(String name, String email, String password) async {
    setState(() {
      _state = _state.copyWith(isLoading: true, error: null);
    });

    try {
      await Future.delayed(const Duration(seconds: 1));

      setState(() {
        _state = AuthState(
          user: UserModel(
            id: '1',
            name: name,
            email: email,
          ),
          isLoading: false,
          isAuthenticated: true,
        );
      });
    } catch (e) {
      setState(() {
        _state = _state.copyWith(
          isLoading: false,
          error: 'Error de registro: $e',
        );
      });
    }
  }

  void logout() {
    setState(() {
      _state = const AuthState();
    });
  }

  @override
  Widget build(BuildContext context) {
    return AuthInherited(
      state: _state,
      login: login,
      register: register,
      logout: logout,
      child: widget.child,
    );
  }
}

    this.todos = const [],
    this.filter = 'all',
  });

  TodoList copyWith({
    List<Todo>? todos,
    String? filter,
  }) {
    return TodoList(
      todos: todos ?? this.todos,
      filter: filter ?? this.filter,
    );
  }

  List<Todo> get filteredTodos {
    switch (filter) {
      case 'active':
        return todos.where((t) => !t.completed).toList();
      case 'completed':
        return todos.where((t) => t.completed).toList();
      default:
        return todos;
    }
  }

  int get activeCount => todos.where((t) => !t.completed).length;
  int get completedCount => todos.where((t) => t.completed).length;
}

// StateNotifier
class TodoNotifier extends StateNotifier<TodoList> {
  TodoNotifier() : super(TodoList());

  void add(String title) {
    state = state.copyWith(
      todos: [
        ...state.todos,
        Todo(
          id: DateTime.now().millisecondsSinceEpoch.toString(),
          title: title,
        ),
      ],
    );
  }

  void toggle(String id) {
    state = state.copyWith(
      todos: state.todos.map((todo) {
        if (todo.id == id) {
          return todo.copyWith(completed: !todo.completed);
        }
        return todo;
      }).toList(),
    );
  }

  void remove(String id) {
    state = state.copyWith(
      todos: state.todos.where((todo) => todo.id != id).toList(),
    );
  }

  void setFilter(String filter) {
    state = state.copyWith(filter: filter);
  }

  void clearCompleted() {
    state = state.copyWith(
      todos: state.todos.where((todo) => !todo.completed).toList(),
    );
  }
}

class CounterBloc extends Bloc<CounterEvent, CounterState> {
  CounterBloc() : super(const CounterState(count: 0)) {
    on<CounterIncrementPressed>(_onIncrement);
    on<CounterDecrementPressed>(_onDecrement);
    on<CounterResetPressed>(_onReset);

  void _onIncrement(CounterIncrementPressed event, Emitter<CounterState> emit) {
    emit(state.copyWith(count: state.count + 1));

  void _onDecrement(CounterDecrementPressed event, Emitter<CounterState> emit) {
    emit(state.copyWith(count: state.count - 1));

class AuthBloc extends Bloc<AuthEvent, AuthState> {
  final AuthRepository _authRepository;

  AuthBloc(this._authRepository) : super(AuthInitial()) {
    on<AuthLoginRequested>(_onLogin);
    on<AuthLogoutRequested>(_onLogout);
    on<AuthStatusChecked>(_onStatusChecked);
  }

  Future<void> _onLogin(AuthLoginRequested event, Emitter<AuthState> emit) async {
    emit(AuthLoading());
    try {
      final user = await _authRepository.login(event.email, event.password);
      emit(Authenticated(user));
    } catch (e) {
      emit(AuthError(e.toString()));
    }
  }

  Future<void> _onLogout(AuthLogoutRequested event, Emitter<AuthState> emit) async {
    emit(AuthLoading());
    try {
      await _authRepository.logout();
      emit(Unauthenticated());
    } catch (e) {
      emit(AuthError(e.toString()));
    }
  }

  Future<void> _onStatusChecked(AuthStatusChecked event, Emitter<AuthState> emit) async {
    emit(AuthLoading());
    try {
      final user = await _authRepository.getCurrentUser();
      if (user != null) {
        emit(Authenticated(user));
      } else {
        emit(Unauthenticated());
      }
    } catch (e) {
      emit(AuthError(e.toString()));
    }
  }
}

class SettingsModel extends ChangeNotifier {
  static const _themeKey = 'theme_mode';
  static const _localeKey = 'locale';
  static const _notificationsKey = 'notifications_enabled';

  ThemeMode _themeMode = ThemeMode.system;
  Locale _locale = const Locale('es');
  bool _notificationsEnabled = true;

  ThemeMode get themeMode => _themeMode;
  Locale get locale => _locale;
  bool get notificationsEnabled => _notificationsEnabled;

  Future<void> load() async {
    final prefs = await SharedPreferences.getInstance();

    _themeMode = _stringToThemeMode(prefs.getString(_themeKey));
    final localeString = prefs.getString(_localeKey);
    if (localeString != null) {
      _locale = Locale(localeString);
    }
    _notificationsEnabled = prefs.getBool(_notificationsKey) ?? true;

    notifyListeners();
  }

  Future<void> setThemeMode(ThemeMode mode) async {
    _themeMode = mode;
    notifyListeners();

    final prefs = await SharedPreferences.getInstance();
    await prefs.setString(_themeKey, _themeModeToString(mode));
  }

  Future<void> setLocale(Locale locale) async {
    _locale = locale;
    notifyListeners();

    final prefs = await SharedPreferences.getInstance();
    await prefs.setString(_localeKey, locale.languageCode);
  }

  Future<void> setNotificationsEnabled(bool enabled) async {
    _notificationsEnabled = enabled;
    notifyListeners();

    final prefs = await SharedPreferences.getInstance();
    await prefs.setBool(_notificationsKey, enabled);
  }

  String _themeModeToString(ThemeMode mode) {
    switch (mode) {
      case ThemeMode.light:
        return 'light';
      case ThemeMode.dark:
        return 'dark';
      default:
        return 'system';
    }
  }

  ThemeMode _stringToThemeMode(String? value) {
    switch (value) {
      case 'light':
        return ThemeMode.light;
      case 'dark':
        return ThemeMode.dark;
      default:
        return ThemeMode.system;
    }
  }
}

abstract class ProductRepository {
  Future<List<Product>> getProducts();
  Future<Product> getProduct(String id);
  Future<void> addProduct(Product product);
  Future<void> updateProduct(Product product);
  Future<void> deleteProduct(String id);
}

// Implementación API
class ApiProductRepository implements ProductRepository {
  final http.Client _client;

  ApiProductRepository(this._client);

  @override
  Future<List<Product>> getProducts() async {
    final response = await _client.get(Uri.parse('https://api.example.com/products'));
    if (response.statusCode != 200) {
      throw Exception('Failed to load products');
    }
    final List<dynamic> data = json.decode(response.body);
    return data.map((json) => Product.fromJson(json)).toList();
  }

  @override
  Future<Product> getProduct(String id) async {
    final response = await _client.get(Uri.parse('https://api.example.com/products/$id'));
    if (response.statusCode != 200) {
      throw Exception('Failed to load product');
    }
    return Product.fromJson(json.decode(response.body));
  }

  @override
  Future<void> addProduct(Product product) async {
    final response = await _client.post(
      Uri.parse('https://api.example.com/products'),
      headers: {'Content-Type': 'application/json'},
      body: json.encode(product.toJson()),
    );
    if (response.statusCode != 201) {
      throw Exception('Failed to add product');
    }
  }

  @override
  Future<void> updateProduct(Product product) async {
    final response = await _client.put(
      Uri.parse('https://api.example.com/products/${product.id}'),
      headers: {'Content-Type': 'application/json'},
      body: json.encode(product.toJson()),
    );
    if (response.statusCode != 200) {
      throw Exception('Failed to update product');
    }
  }

  @override
  Future<void> deleteProduct(String id) async {
    final response = await _client.delete(Uri.parse('https://api.example.com/products/$id'));
    if (response.statusCode != 204) {
      throw Exception('Failed to delete product');
    }
  }
}

class TodoListNotifier extends StateNotifier<List<Todo>> {
  TodoListNotifier() : super([]);

  void add(String title) {
    state = [
      ...state,
      Todo(id: DateTime.now().millisecondsSinceEpoch.toString(), title: title),
    ];
  }

  void toggle(String id) {
    state = state.map((todo) {
      if (todo.id == id) {
        return todo.copyWith(completed: !todo.completed);
      }
      return todo;
    }).toList();
  }

  void remove(String id) {
    state = state.where((todo) => todo.id != id).toList();
  }

  void clearCompleted() {
    state = state.where((todo) => !todo.completed).toList();
  }

  void edit(String id, String newTitle) {
    state = state.map((todo) {
      if (todo.id == id) {
        return todo.copyWith(title: newTitle);
      }
      return todo;
    }).toList();
  }
}

Conclusión

La gestión de estado en Flutter ofrece múltiples soluciones según las necesidades:

  1. setState: Para estado local simple y temporal

  2. InheritedWidget: Para estado compartido sin dependencias externas

  3. Provider: Para estado de aplicación simple-medio con fácil aprendizaje

  4. Riverpod: Para estado complejo, asíncrono y mejor testing

  5. BLoC/Cubit: Para proyectos grandes con flujos complejos

  6. GetX: Para desarrollo rápido con todo incluido

Recomendaciones: - Empezar con Provider o Riverpod para la mayoría de proyectos - Considerar BLoC para equipos grandes y testing extensivo - Usar setState solo para UI state local - Mantener estados inmutables - Separar siempre la lógica de la UI - Implementar persistencia cuando sea necesario
